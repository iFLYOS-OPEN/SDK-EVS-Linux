
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DHAVE_PTHREADS -DHAVE_SYS_UIO_H -DHAVE_ENDIAN_H -std=c++11")
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -DHAVE_PTHREADS -DHAVE_SYS_UIO_H -DHAVE_ENDIAN_H")

aux_source_directory(. DIR_SRCS)
list(REMOVE_ITEM DIR_SRCS ./ashmem-host.c ./ashmem-dev.c ./properties.c)
add_library(android_cutils SHARED ${DIR_SRCS})
target_link_libraries(android_cutils pthread )#property)

INSTALL( TARGETS android_cutils ${INSTALL_EXPORT}
     RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
     LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
     ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR})
